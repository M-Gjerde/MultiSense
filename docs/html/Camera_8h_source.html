28 void updateViewMatrix()
29 {
30 glm::mat4 rotM = glm::mat4(1.0f);
31 glm::mat4 transM;
32
33 rotM = glm::rotate(rotM, glm::radians(rotation.x * (flipY ? -1.0f : 1.0f)), glm::vec3(1.0f, 0.0f, 0.0f));
34 rotM = glm::rotate(rotM, glm::radians(rotation.y), glm::vec3(0.0f, 1.0f, 0.0f));
35 rotM = glm::rotate(rotM, glm::radians(rotation.z), glm::vec3(0.0f, 0.0f, 1.0f));
36
37 glm::vec3 translation = position;
38 if (flipY) {
39 translation.y *= -1.0f;
40 }
41 transM = glm::translate(glm::mat4(1.0f), translation);
42
43 if (type == CameraType::firstperson)
44 {
45 matrices.view = rotM * transM;
46 }
47 else
48 {
49 matrices.view = transM * rotM;
50 }
51
52 viewPos = glm::vec4(position, 0.0f) * glm::vec4(-1.0f, 1.0f, -1.0f, 1.0f);
53
54 updated = true;
55 };

157 void update(float deltaTime)
158 {
159 updated = false;
160 if (type == CameraType::firstperson || type == CameraType::lookat)
161 {
162 if (moving())
163 {
164
165 glm::vec3 camFront;
166 camFront.x = -cos(glm::radians(rotation.x)) * sin(glm::radians(rotation.y));
167 camFront.y = sin(glm::radians(rotation.x));
168 camFront.z = cos(glm::radians(rotation.x)) * cos(glm::radians(rotation.y));
169 camFront = glm::normalize(camFront);
170
171 float moveSpeed = deltaTime * movementSpeed;
172
173 if (keys.up)
174 position += camFront * moveSpeed;
175 if (keys.down)
176 position -= camFront * moveSpeed;
177 if (keys.left)
178 position -= glm::normalize(glm::cross(camFront, glm::vec3(0.0f, 1.0f, 0.0f))) * moveSpeed;
179 if (keys.right)
180 position += glm::normalize(glm::cross(camFront, glm::vec3(0.0f, 1.0f, 0.0f))) * moveSpeed;
181
182 updateViewMatrix();
183 }
184 }

187 // Update camera passing separate axis data (gamepad)
188 // Returns true if view or position has been changed
189 bool updatePad(glm::vec2 axisLeft, glm::vec2 axisRight, float deltaTime)
190 {
191 bool retVal = false;
192
193 if (type == CameraType::firstperson)
194 {
195 // Use the common console thumbstick layout
196 // Left = view, right = move
197
198 const float deadZone = 0.0015f;
199 const float range = 1.0f - deadZone;
200
201 glm::vec3 camFront;
202 camFront.x = -cos(glm::radians(rotation.x)) * sin(glm::radians(rotation.y));
203 camFront.y = sin(glm::radians(rotation.x));
204 camFront.z = cos(glm::radians(rotation.x)) * cos(glm::radians(rotation.y));
205 camFront = glm::normalize(camFront);
206
207 float moveSpeed = deltaTime * movementSpeed * 2.0f;
208 float rotSpeed = deltaTime * rotationSpeed * 50.0f;
209
210 // Move
211 if (fabsf(axisLeft.y) > deadZone)
212 {
213 float pos = (fabsf(axisLeft.y) - deadZone) / range;
214 position -= camFront * pos * ((axisLeft.y < 0.0f) ? -1.0f : 1.0f) * moveSpeed;
215 retVal = true;
216 }
217 if (fabsf(axisLeft.x) > deadZone)
218 {
219 float pos = (fabsf(axisLeft.x) - deadZone) / range;
220 position += glm::normalize(glm::cross(camFront, glm::vec3(0.0f, 1.0f, 0.0f))) * pos * ((axisLeft.x < 0.0f) ? -1.0f : 1.0f) * moveSpeed;
221 retVal = true;
222 }
223
224 // Rotate
225 if (fabsf(axisRight.x) > deadZone)
226 {
227 float pos = (fabsf(axisRight.x) - deadZone) / range;
228 rotation.y += pos * ((axisRight.x < 0.0f) ? -1.0f : 1.0f) * rotSpeed;
229 retVal = true;
230 }
231 if (fabsf(axisRight.y) > deadZone)
232 {
233 float pos = (fabsf(axisRight.y) - deadZone) / range;
234 rotation.x -= pos * ((axisRight.y < 0.0f) ? -1.0f : 1.0f) * rotSpeed;
235 retVal = true;
236 }
237 }
238 else
239 {
240 // todo: move code from example base class for look-at
241 }
242
243 if (retVal)
244 {
245 updateViewMatrix();
246 }
247
248 return retVal;
249 }

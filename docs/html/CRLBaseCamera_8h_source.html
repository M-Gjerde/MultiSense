51 PointCloudData(uint32_t width, uint32_t height) {
52 vertexCount = width * height;
53 // Virtual class can generate some mesh data here
54 vertices = calloc(vertexCount, sizeof(MeshModel::Model::Vertex));
55
56 uint32_t v = 0;
57 auto *vP = (MeshModel::Model::Vertex *) vertices;
58 for (uint32_t x = 0; x < width; ++x) {
59 for (uint32_t z = 0; z < height; ++z) {
60 MeshModel::Model::Vertex vertex{};
61 vertex.pos = glm::vec3((float) x / 50, 0.0f, (float) z / 50);
62 vertex.uv0 = glm::vec2((float) x / (float) width, (float) z / (float) height);
63 vP[v] = vertex;
64 v++;
65 }
66 }
67 };
68
69 ~PointCloudData() {
70 free(vertices);
71 delete[] indices;
72 }

74 };

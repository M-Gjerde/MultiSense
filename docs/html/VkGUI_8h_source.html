146 // Create font texture
147 unsigned char *fontData;
148 int texWidth, texHeight;
149 io.Fonts->GetTexDataAsRGBA32(&fontData, &texWidth, &texHeight);
150 VkDeviceSize uploadSize = texWidth * texHeight * 4 * sizeof(char);
151
152 // Create Target Image for copy
153 VkImageCreateInfo imageInfo = Populate::imageCreateInfo();
154 imageInfo.imageType = VK_IMAGE_TYPE_2D;
155 imageInfo.format = VK_FORMAT_R8G8B8A8_UNORM;
156 imageInfo.extent.width = texWidth;
157 imageInfo.extent.height = texHeight;
158 imageInfo.extent.depth = 1;
159 imageInfo.mipLevels = 1;
160 imageInfo.arrayLayers = 1;
161 imageInfo.samples = VK_SAMPLE_COUNT_1_BIT;
162 imageInfo.tiling = VK_IMAGE_TILING_OPTIMAL;
163 imageInfo.usage = VK_IMAGE_USAGE_SAMPLED_BIT | VK_IMAGE_USAGE_TRANSFER_DST_BIT;
164 imageInfo.sharingMode = VK_SHARING_MODE_EXCLUSIVE;
165 imageInfo.initialLayout = VK_IMAGE_LAYOUT_UNDEFINED;
166 if (vkCreateImage(device->logicalDevice, &imageInfo, nullptr, &fontImage) != VK_SUCCESS)
167 throw std::runtime_error("Failed to create ImGUI image");
168 VkMemoryRequirements memReqs;
169 vkGetImageMemoryRequirements(device->logicalDevice, fontImage, &memReqs);
170 VkMemoryAllocateInfo memAllocInfo = Populate::memoryAllocateInfo();
171 memAllocInfo.allocationSize = memReqs.size;
172 memAllocInfo.memoryTypeIndex = device->getMemoryType(memReqs.memoryTypeBits,
173 VK_MEMORY_PROPERTY_DEVICE_LOCAL_BIT);
174 if (vkAllocateMemory(device->logicalDevice, &memAllocInfo, nullptr, &fontMemory) != VK_SUCCESS)
175 throw std::runtime_error("Failed to allocate image IMGUI memory");
176 if (vkBindImageMemory(device->logicalDevice, fontImage, fontMemory, 0))
177 throw std::runtime_error("Failed to bind imgui image with its memory");
178
179 // Create Image view to interface with image.
180 // Image view
181 VkImageViewCreateInfo viewInfo = Populate::imageViewCreateInfo();
182 viewInfo.image = fontImage;
183 viewInfo.viewType = VK_IMAGE_VIEW_TYPE_2D;
184 viewInfo.format = VK_FORMAT_R8G8B8A8_UNORM;
185 viewInfo.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
186 viewInfo.subresourceRange.levelCount = 1;
187 viewInfo.subresourceRange.layerCount = 1;
188 if (vkCreateImageView(device->logicalDevice, &viewInfo, nullptr, &fontView) != VK_SUCCESS)
189 throw std::runtime_error("Failed to create image view for imgui image");
190 // Staging buffers for font upload
191 Buffer stagingBuffer;
192 // Copy data to staging buffers
193 if (device->createBuffer(VK_BUFFER_USAGE_TRANSFER_SRC_BIT,
194 VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT,
195 &stagingBuffer, uploadSize) != VK_SUCCESS)
196 throw std::runtime_error("Failed to create staging buffer");
197 stagingBuffer.map();
198 memcpy(stagingBuffer.mapped, fontData, uploadSize);
199 stagingBuffer.unmap();
200
201 // Transition image layout to transfer and specify copy region
202 // Copy buffer data to font image
203 VkCommandBuffer copyCmd = device->createCommandBuffer(VK_COMMAND_BUFFER_LEVEL_PRIMARY, true);
204
205 // Prepare for transfer
206 Utils::setImageLayout(
207 copyCmd,
208 fontImage,
209 VK_IMAGE_ASPECT_COLOR_BIT,
210 VK_IMAGE_LAYOUT_UNDEFINED,
211 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
212 VK_PIPELINE_STAGE_HOST_BIT,
213 VK_PIPELINE_STAGE_TRANSFER_BIT);
214
215 // Copy
216 // Copy staging buffer to image
217 Utils::copyBufferToImage(copyCmd, stagingBuffer.buffer, fontImage, texWidth, texHeight,
218 VK_IMAGE_ASPECT_COLOR_BIT);
219
220 // Prepare image for shader read
221 // Prepare for shader read
222 Utils::setImageLayout(
223 copyCmd,
224 fontImage,
225 VK_IMAGE_ASPECT_COLOR_BIT,
226 VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL,
227 VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL,
228 VK_PIPELINE_STAGE_TRANSFER_BIT,
229 VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT);
230
231 device->flushCommandBuffer(copyCmd, copyQueue, true);
232
233 stagingBuffer.destroy();
234 // submit cmd buffer and destroy staging buffer

236 // Texture sampler
237 // Font texture Sampler
238 VkSamplerCreateInfo samplerInfo = Populate::samplerCreateInfo();
239 samplerInfo.magFilter = VK_FILTER_LINEAR;
240 samplerInfo.minFilter = VK_FILTER_LINEAR;
241 samplerInfo.mipmapMode = VK_SAMPLER_MIPMAP_MODE_LINEAR;
242 samplerInfo.addressModeU = VK_SAMPLER_ADDRESS_MODE_CLAMP_TO_EDGE;
243 samplerInfo.addressModeV = VK_SAMPLER_ADDRESS_MODE_CLAMP_TO_EDGE;
244 samplerInfo.addressModeW = VK_SAMPLER_ADDRESS_MODE_CLAMP_TO_EDGE;
245 samplerInfo.borderColor = VK_BORDER_COLOR_FLOAT_OPAQUE_WHITE;
246 if (vkCreateSampler(device->logicalDevice, &samplerInfo, nullptr, &sampler) != VK_SUCCESS)
247 throw std::runtime_error("Failed to create texture sampler IMGUI");
248 // Descriptor pool
249 // Descriptor layout
250 // Descriptor set
251
252 // Descriptor pool
253 std::vector<VkDescriptorPoolSize> poolSizes = {
254 Populate::descriptorPoolSize(VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER, 1)
255 };
256 VkDescriptorPoolCreateInfo descriptorPoolInfo = Populate::descriptorPoolCreateInfo(poolSizes, 2);

263 Populate::descriptorSetLayoutBinding(VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER,

268 VK_SUCCESS)
269 throw std::runtime_error("Failed to create descriptor set layout");
270
271 // Descriptor set
272 VkDescriptorSetAllocateInfo allocInfo = Populate::descriptorSetAllocateInfo(descriptorPool,
273 &descriptorSetLayout, 1);
274 if (vkAllocateDescriptorSets(device->logicalDevice, &allocInfo, &descriptorSet) != VK_SUCCESS)
275 throw std::runtime_error("Failed to create descriptor set");
276 VkDescriptorImageInfo fontDescriptor = Populate::descriptorImageInfo(
277 sampler,
278 fontView,
279 VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL
280 );
281 std::vector<VkWriteDescriptorSet> writeDescriptorSets = {
282 Populate::writeDescriptorSet(descriptorSet, VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER, 0,
283 &fontDescriptor)
284 };
285 vkUpdateDescriptorSets(device->logicalDevice, static_cast<uint32_t>(writeDescriptorSets.size()),
286 writeDescriptorSets.data(), 0, nullptr);

539// Update vertex and index buffer containing the imGui elements when required
540 bool updateBuffers() {
541 ImDrawData *imDrawData = ImGui::GetDrawData();
542 bool updateCommandBuffers = false;
543
544 // Note: Alignment is done inside buffer creation
545 VkDeviceSize vertexBufferSize = imDrawData->TotalVtxCount * sizeof(ImDrawVert);
546 VkDeviceSize indexBufferSize = imDrawData->TotalIdxCount * sizeof(ImDrawIdx);
547
548 if ((vertexBufferSize == 0) || (indexBufferSize == 0)) {
549 return false;
550 }
551
552 // Update buffers only if vertex or index count has been changed compared to current buffer size
553
554 // Vertex buffer
555 if ((vertexBuffer.buffer == VK_NULL_HANDLE) || (vertexCount != imDrawData->TotalVtxCount)) {
556 vertexBuffer.unmap();
557 vertexBuffer.destroy();
558 if (VK_SUCCESS !=
559 device->createBuffer(VK_BUFFER_USAGE_VERTEX_BUFFER_BIT, VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT,
560 &vertexBuffer, vertexBufferSize))
561 throw std::runtime_error("Failed to create vertex Buffer");
562 vertexCount = imDrawData->TotalVtxCount;
563 vertexBuffer.map();
564 updateCommandBuffers = true;
565 }
566
567 // Index buffer
568 if ((indexBuffer.buffer == VK_NULL_HANDLE) || (indexCount < imDrawData->TotalIdxCount)) {
569 indexBuffer.unmap();
570 indexBuffer.destroy();
571 if (VK_SUCCESS !=
572 device->createBuffer(VK_BUFFER_USAGE_INDEX_BUFFER_BIT, VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT,
573 &indexBuffer, indexBufferSize))
574 throw std::runtime_error("Failed to create index buffer");
575 indexCount = imDrawData->TotalIdxCount;
576 indexBuffer.map();
577 updateCommandBuffers = true;
578 }
579
580 // Upload data
581 ImDrawVert *vtxDst = (ImDrawVert *) vertexBuffer.mapped;
582 ImDrawIdx *idxDst = (ImDrawIdx *) indexBuffer.mapped;
583
584 for (int n = 0; n < imDrawData->CmdListsCount; n++) {
585 const ImDrawList *cmd_list = imDrawData->CmdLists[n];
586 memcpy(vtxDst, cmd_list->VtxBuffer.Data, cmd_list->VtxBuffer.Size * sizeof(ImDrawVert));
587 memcpy(idxDst, cmd_list->IdxBuffer.Data, cmd_list->IdxBuffer.Size * sizeof(ImDrawIdx));
588 vtxDst += cmd_list->VtxBuffer.Size;
589 idxDst += cmd_list->IdxBuffer.Size;
590 }
591
592 // Flush to make writes visible to GPU
593 vertexBuffer.flush();
594 indexBuffer.flush();
595
596 return updateCommandBuffers;
597 }

638 scissorRect.extent.height = (uint32_t) (pcmd->ClipRect.w - pcmd->ClipRect.y);
639 vkCmdSetScissor(commandBuffer, 0, 1, &scissorRect);
